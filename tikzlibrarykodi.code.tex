\usetikzlibrary[decorations.markings]
\usetikzlibrary[arrows.meta]
\usetikzlibrary[matrix]

%\usetikzlibrary[kodi.bugfixes]

%\usetikzlibrary[kodi.balancer]

\def\startkodi{\starttikzpicture[kodi]}
\def\stopkodi{\stoptikzpicture}

\let\stopcommutativediagram\stopkodi
